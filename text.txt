    <textarea id="noteArea" placeholder="ل ڤێرێ بنڤێسە..."></textarea>
    <br>
    <button class="btn" onclick="saveNote()">Save Note</button>
    <p id="status"></p>

    <script>
        // بو خەزنکرنا نڤێسێ د ناڤ موبایلێ دا
        function saveNote() {
            const note = document.getElementById('noteArea').value;
            localStorage.setItem('userNote', note);
            document.getElementById('status').innerText = "تێبینی هاتە پاراستن!";
            setTimeout(() => { document.getElementById('status').innerText = ""; }, 2000);
        }

        // بۆ نیشاندانا نڤێسێ دەما بەرنامە ڤەدبیت
        window.onload = function() {
            const savedNote = localStorage.getItem('userNote');
            if (savedNote) {
                document.getElementById('noteArea').value = savedNote;
            }
        }
    </script>
